R(Оё)=[cosвҒЎОёвҲ’sinвҒЎОёsinвҒЎОёcosвҒЎОё]=cosвҒЎОё[1001]+sinвҒЎОё[0вҲ’110]=expвҒЎ(Оё[0вҲ’110]) R(\theta) = \begin{bmatrix} \cos \theta & -\sin \theta \\ \sin \theta & \cos \theta \end{bmatrix} = \cos \theta \begin{bmatrix} 1 & 0 \\ 0 & 1 \end{bmatrix} + \sin \theta \begin{bmatrix} 0 & -1 \\ 1 & 0 \end{bmatrix} = \exp \left( \theta \begin{bmatrix} 0 & -1 \\ 1 & 0 \end{bmatrix} \right) R(Оё)=[cosОёsinОёвҖӢвҲ’sinОёcosОёвҖӢ]=cosОё[10вҖӢ01вҖӢ]+sinОё[01вҖӢвҲ’10вҖӢ]=exp(Оё[01вҖӢвҲ’10вҖӢ])

зү№еҫҒеҖј

дәҢз»ҙж—ӢиҪ¬зҹ©йҳөжңүдёҖеҜ№е…ұиҪӯеӨҚзү№еҫҒеҖј, жІЎжңүе®һзү№еҫҒеҖј

О»1=eiОё,О»2=eвҲ’iОё \lambda_1 = e^{i\theta}, \quad \lambda_2 = e^{-i\theta} О»1вҖӢ=eiОё,О»2вҖӢ=eвҲ’iОё

detвҒЎ(R(Оё)вҲ’О»I)=0 \det(R(\theta) - \lambda I) = 0 det(R(Оё)вҲ’О»I)=0

detвҒЎ(cosвҒЎОёвҲ’О»вҲ’sinвҒЎОёsinвҒЎОёcosвҒЎОёвҲ’О»)=0 \det \begin{pmatrix} \cos\theta - \lambda & -\sin\theta \\ \sin\theta & \cos\theta - \lambda \end{pmatrix} = 0 det(cosОёвҲ’О»sinОёвҖӢвҲ’sinОёcosОёвҲ’О»вҖӢ)=0

(cosвҒЎОёвҲ’О»)2+sinвҒЎ2Оё=0 (\cos\theta - \lambda)^2 + \sin^2\theta = 0 (cosОёвҲ’О»)2+sin2Оё=0

cosвҒЎ2ОёвҲ’2О»cosвҒЎОё+О»2+sinвҒЎ2Оё=0 \cos^2\theta - 2\lambda\cos\theta + \lambda^2 + \sin^2\theta = 0 cos2ОёвҲ’2О»cosОё+О»2+sin2Оё=0
